class Solution:

    def maxSubArray(self, nums):
        curr_sum = nums[0]
        max_sum = nums[0]

        for num in nums[1:]:
            curr_sum = max(num, curr_sum + num)
            max_sum = max(max_sum, curr_sum)

        return max_sum


def main():
    nums = [-2, 1, -3, 4, -1, 2, 1, -5, 4]

    obj = Solution()              # object creation
    result = obj.maxSubArray(nums) # function calling

    print("Maximum Subarray Sum is:", result)


# Calls main function
if __name__ == "__main__":
    main()
