class Solution:

    def productExceptSelf(self, nums):

        n = len(nums)
        answer = [1] * n

        # LEFT pass
        left = 1
        for i in range(n):
            answer[i] = left
            left *= nums[i]

        # RIGHT pass
        right = 1
        for i in range(n-1, -1, -1):
            answer[i] *= right
            right *= nums[i]

        return answer


def main():

    nums = [1,2,3,4]

    obj = Solution()
    result = obj.productExceptSelf(nums)

    print(result)


if __name__ == "__main__":
    main()

I compute prefix (left) products in one pass and suffix (right) products in another pass. Multiplying them gives the result without division.
